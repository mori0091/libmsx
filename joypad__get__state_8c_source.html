67uint8_t joypad_get_state(uint8_t controller) {
68 uint8_t joy;
69 switch (controller) {
70 case 0:
71 joy = lot[msx_GTSTCK(0)];
72 joy |= msx_GTTRIG(0) & VK_FIRE_0;
73 return joy;
74 case 1:
75 joy = lot[msx_GTSTCK(1)];
76 joy |= msx_GTTRIG(1) & VK_FIRE_0;

78 return joy;
