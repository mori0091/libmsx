333static void sound_channel_output(void) {

335 if (x < 0) {

343static void sound_player__process_channel(void) {
344 // ---- Reads head (chunk size and next duration) ----
345 uint8_t len;
346 const uint8_t * next = ctx.stch->next;
347 {
348 const uint8_t head1 = *next++;
349 if (head1 == 255) {
350 // end of music track
351 ctx.st->flag &= ~ctx.mask;
352 return;
353 }
354 const uint8_t ticks_lo = *next++;
355 const uint8_t ticks_hi = (head1 & 0x1f);
356 const uint16_t ticks = (ticks_hi << 8) | (ticks_lo);
357 ctx.stch->duration += SOUND_SPEED_1X * ticks;
358 len = (head1 >> 5) & 7;
359 ctx.stch->next = next + len;
360 }
361 // ---- Parse and process the chunk ----
362 while (len--) {
363 const uint8_t x = *next++;
364 switch (x >> 4) {
365 case 0x0:
366 // keyon (12bits)
367 // 0000b fdr_hi:4 fdr_lo:8
368 // -- Changes frequency of tone generator and performs a key-on.
369 sound_eg_key_on(&ctx.stch->eg);
370 // When using hardware envelopes, special processing is required.
371 if (ctx.stch->hw_envelope_enable) {
372 if (ctx.channel_muted) break;
373 PSG_SET(13, ctx.st->envelope_pattern);
374 PSG_SET(8+ctx.ch, 16);
375 }
376 // [[FALL THROUGH]]
377 case 0xe:
378 // tone (12bits)
379 // 1110b fdr_hi:4 fdr_lo:8
380 // -- changes frequency of tone generator but do not key-on.
381 {
382 const uint8_t fdr_hi = x & 0x0f;
383 const uint8_t fdr_lo = *next++;
384 len--;
385 if (ctx.channel_muted) break;
386 PSG_SET(1+2*ctx.ch, fdr_hi); // fdr_hi:4
387 PSG_SET(0+2*ctx.ch, fdr_lo); // fdr_lo:8
388 }
389 break;
390 case 0x1:
391 // keyoff (0bits)
392 // -- Perform a key-off. (It also serves as a rest.)
393 sound_eg_key_off(&ctx.stch->eg);
394 if (ctx.stch->hw_envelope_enable) {
395 if (ctx.channel_muted) break;
396 PSG_SET(8+ctx.ch, 0);
397 }
398 break;
399 case 0x2:
400 case 0x3:
401 // noise (5bits)
402 // 001b fdr:5
403 // -- Changes frequency of noise generator.
404 if (ctx.channel_muted) break;
405 PSG_SET(6, (x & 0x1f));
406 break;
407 case 0x7:
408 // mixer (6bits)
409 // 0111b ----b --b NC:1 NB:1 NA:1 TC:1 TB:1 TA:1
410 // -- Changes the mixer setting.
411 {
412 len--;
413 // const uint8_t mixer = (*next++ & 0x3f) | 0x80;
414 const uint8_t mixer = *next++;
415 if (ctx.channel_muted) break;
416 PSG_SET(7, mixer);
417 }
418 break;
419 case 0x8:
420 // volume (4bits)
421 // 1000b vol:4
422 // -- Changes the amplifier level.
423 {
424 ctx.stch->amp = x & 0x0f;
425 ctx.stch->hw_envelope_enable = false;
426 }
427 break;
428 case 0xc:
429 // software envelope pattern (4bits)
430 // 1100b pat:4
431 // -- Set the software envelope pattern.
432 {
433 sound_channel_set_envelope(ctx.stch, x & 0x0f);
434 ctx.stch->hw_envelope_enable = false;
435 }
436 break;
437 case 0x4:
438 case 0x9:
439 case 0xd:
440 if (x & 0x90) {
441 // envelope pattern (4bits)
442 // 1-01b pat:4
443 // -- Set the hardware envelope pattern, and enable hardware envelope.
444 // The envelope pattern and switch states are saved (not applied to
445 // the registers here) and applied at each "key on" timing.
446 ctx.st->envelope_pattern = x & 0x0f;
447 ctx.stch->hw_envelope_enable = true;
448 }
449 if (x & 0x40) {
450 // envelope cycle (16bits)
451 // -10-b ----b fdr_hi:8 fdr_lo:8
452 // -- Set the hardware envelope cycle.
453 const uint8_t cycle_hi = *next++;
454 const uint8_t cycle_lo = *next++;
455 len -= 2;
456 if (ctx.channel_muted) break;
457 PSG_SET(12, cycle_hi); // cycle_hi:8
458 PSG_SET(11, cycle_lo); // cycle_lo:8
459 }
460 break;
461 default:
462 // stream error
463 len = 0;
464 ctx.st->flag &= ~ctx.mask;
465 break;
466 }
467 }
468}

470static bool sound_player__process(void) {
471 while (ctx.st->section < ctx.st->clip->num_fragments) {
472 {
473 const uint8_t count = ctx.st->count_per_tick * sound_advance_count;
474 ctx.mask = SOUND_CHANNEL_A;
475 for (ctx.ch = 0; ctx.ch < 3; ++ctx.ch, ctx.mask <<= 1) {
476 if (ctx.st->flag & ctx.mask) {
477 ctx.channel_muted = (bool)(ctx.mute & ctx.mask);
478 ctx.stch = &ctx.st->channels[ctx.ch];
479 // ---- Countdown time remaining ----
480 if (count < ctx.stch->duration) {
481 ctx.stch->duration -= count;
482 } else {
483 ctx.stch->duration = 0;
484 // ---- Decode next chunk ----
485 sound_player__process_channel();
486 }
487 if (ctx.channel_muted || ctx.stch->hw_envelope_enable) {
488 continue;
489 }
490 sound_channel_output();
491 }
492 }
493 }
494 if (ctx.st->flag & SOUND_CHANNEL_ALL) {
495 // current fragment is not finished.
496 return false;
497 }
498 if (++ctx.st->section >= ctx.st->clip->num_fragments) {
499 // end of the clip
500 break;
501 }
502 // the clip has more fragments
503 const struct sound_fragment * sf = ctx.st->clip->fragments[ctx.st->section];
504 if (!sf) {
505 break;
506 }
507 // set the next fragment
508 sound_set_fragment(ctx.st, sf);
509 if (!(ctx.st->flag & SOUND_CHANNEL_ALL)) {
510 // end of the clip (invalid fragment)
511 break;
512 }
513 // start to play the next fragment.
514 }
515 ctx.st->section = ctx.st->clip->num_fragments;
516 return true;
517}

520void sound_player(void) {
521 if (sound.paused) {
522 return;
523 }
524 ay_3_8910_play();
525 sound_effect_apply();
526 sound_update_tick();
527 // ---- background music ----
528 if (sound.bg.clip) {
529 // By passing `mute` as 2nd parameter, temporarily turn on the mute switch
530 // for the channel being used for sound effects.
531 ctx.st = &sound.bg;
532 ctx.mute = ((sound.bg.flag >> 3) | sound.se.flag) & 7;
533 if (sound_player__process()) {
534 // ---- end of music ----
535 if (sound.repeat) {
536 // ---- auto-repeat ----
537 sound_set_clip(&sound.bg, sound.bg.clip);
538 }
539 else {
540 sound_state_init(&sound.bg);
541 }
542 }
543 }
544 // ---- sound effect ----
545 if (sound.se.clip) {
546 ctx.st = &sound.se;
547 ctx.mute = (sound.se.flag >> 3) & 7;
548 if (sound_player__process()) {
549 // ---- end of music ----
550 sound_state_init(&sound.se);
551 sound_restore_psg_registers();
552 }
553 }
554}
