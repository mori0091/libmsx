97  if (sound.paused) return;
98  if (!(sound.flag & SOUND_CHANNEL_ALL)) {
99  if (!sound.repeat) {
100  sound_stop();
101  return;
102  }
103  sound_set_bgm(sound.data);
104  sound_start();
105  }
106  uint8_t mask = SOUND_CHANNEL_A;
107  for (uint8_t ch = 0; ch < 3; ++ch, mask <<= 1) {
108  // ---- Countdown time remaining ----
109  if (0 < sound.duration[ch]) {
110  sound.duration[ch] -= COUNT_PER_TICK;
111  if (0 < sound.duration[ch]) {
112  continue;
113  }
114  }
115  // ---- Check to see if the stream has finished. ----
116  if (!(sound.flag & mask)) continue; // no more chunk
117  uint8_t head1 = *sound.next[ch]++;
118  if (head1 == 255) {
119  // end of music track
120  sound.flag &= ~mask;
121  continue;
122  }
123  // ---- Reads head (chunk size and next duration) ----
124  uint8_t len = (head1 >> 5) & 7;
125  if (!len) len = 8; // 'length = 0' means 8 bytes chunk
126  const uint16_t ticks = ((head1 & 0x1f) << 8) | (*sound.next[ch]++);
127  sound.duration[ch] += COUNT_PER_TICK_60HZ * ticks;
128  // ---- Parse and process the chunk ----
129  bool muted = sound.flag & (mask << 3);
130  while (len) {
131  len--;
132  uint8_t x = *sound.next[ch]++;
133  uint8_t x_lo = x & 0x0f;
134  switch (x & 0xf0) {
135  case 0x00:
136  // tone (12bits)
137  // 0000b fdr_hi:4 fdr_lo:8
138  sound_psg_set(muted, 1+2*ch, x_lo); // fdr_hi:4
139  sound_psg_set(muted, 0+2*ch, *sound.next[ch]++); // fdr_lo:8
140  len--;
141  break;
142  case 0x10:
143  // rest (0bits)
144  sound_psg_set(muted, 8+ch, 0);
145  break;
146  case 0x20:
147  case 0x30:
148  // noise (5bits)
149  // 001b fdr:5
150  sound_psg_set(muted, 6, x & 0x1f);
151  break;
152  case 0x70:
153  // mixer (6bits)
154  // 0111b ----b --b NC:1 NB:1 NA:1 TC:1 TB:1 TA:1
155  sound_psg_set(muted, 7, (*sound.next[ch]++ & 0x3f) | 0x80);
156  break;
157  case 0x80:
158  // volume (4bits)
159  // 1000b vol:4
160  sound_psg_set(muted, 8+ch, x_lo);
161  break;
162  case 0x40:
163  case 0x90:
164  case 0xd0:

170  }
