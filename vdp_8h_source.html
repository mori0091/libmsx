531struct vdp_cmd {
532 uint8_t r32; // SX (lo)
533 uint8_t r33; // SX (hi)
534 uint8_t r34; // SY (lo)
535 uint8_t r35; // SY (hi)
536 uint8_t r36; // DX (lo)
537 uint8_t r37; // DX (hi)
538 uint8_t r38; // DY (lo)
539 uint8_t r39; // DY (hi)
540 uint8_t r40; // NX (lo)
541 uint8_t r41; // NX (hi)
542 uint8_t r42; // NY (lo)
543 uint8_t r43; // NY (hi)
544 uint8_t r44; // CLR (2bpp * 4 pix, 4bpp * 2 pix, or 8bpp * 1 pix)
545 uint8_t r45; // ARG (0000b | DIY | DIX | 00b)
546 uint8_t r46; // CMR (op | logop)
547};
