537struct vdp_cmd {
538 uint8_t r32; // SX (lo)
539 uint8_t r33; // SX (hi)
540 uint8_t r34; // SY (lo)
541 uint8_t r35; // SY (hi)
542 uint8_t r36; // DX (lo)
543 uint8_t r37; // DX (hi)
544 uint8_t r38; // DY (lo)
545 uint8_t r39; // DY (hi)
546 uint8_t r40; // NX (lo)
547 uint8_t r41; // NX (hi)
548 uint8_t r42; // NY (lo)
549 uint8_t r43; // NY (hi)
550 uint8_t r44; // CLR (2bpp * 4 pix, 4bpp * 2 pix, or 8bpp * 1 pix)
551 uint8_t r45; // ARG (0000b | DIY | DIX | 00b)
552 uint8_t r46; // CMR (op | logop)
553};
