497struct vdp_cmd {
498 uint8_t r32; // SX (lo)
499 uint8_t r33; // SX (hi)
500 uint8_t r34; // SY (lo)
501 uint8_t r35; // SY (hi)
502 uint8_t r36; // DX (lo)
503 uint8_t r37; // DX (hi)
504 uint8_t r38; // DY (lo)
505 uint8_t r39; // DY (hi)
506 uint8_t r40; // NX (lo)
507 uint8_t r41; // NX (hi)
508 uint8_t r42; // NY (lo)
509 uint8_t r43; // NY (hi)
510 uint8_t r44; // CLR (2bpp * 4 pix, 4bpp * 2 pix, or 8bpp * 1 pix)
511 uint8_t r45; // ARG (0000b | DIY | DIX | 00b)
512 uint8_t r46; // CMR (op | logop)
513};
